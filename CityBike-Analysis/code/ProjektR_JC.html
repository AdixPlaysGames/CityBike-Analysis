def getHourDist(DataFrame, day, color = "blue", hour = 6):
    df = DataFrame[(DataFrame["day of week"] == day) & (DataFrame["hour"] >= hour)]
    df = df['hour'].value_counts().to_frame()
    df.columns = ['count']
    df['hour'] = df.index
    df = df.reset_index()
    df = df.drop(columns=['index'])
    df.columns = ['count', 'hour']
    fig = plt.figure(figsize=(20,3))
    ax = fig.add_axes([0,0,1,1])
    langs = df["hour"]
    students = df["count"]
    ax.bar(langs,students, color= color)
    plt.show()

def getAgg(DataFrame, day, hour = 6):
    df = DataFrame[(DataFrame["day of week"] == day) & (DataFrame["hour"] >= hour)]
    df = df['hour'].value_counts().to_frame()
    df.columns = ['count']
    df['hour'] = df.index
    df = df.sort_values(by=['hour'])
    df = df.reset_index()
    df = df.drop(columns=['index'])
    df.columns = ['count', 'hour']
    return df

def CasVsMem(DataFrame, day, hour = 6):

    subscribers.columns = ["subscribers count", "hour"]
    casuals = getAgg(jcSummerDF[jcSummerDF.member_casual == "casual"], day)
    casuals.columns = ["casuals count", "hour"]
    ax = subscribers.plot(x="hour", y="subscribers count", legend=False, figsize=(20,4))
    ax2 = ax.twinx()
    casuals.plot(x= "hour", y="casuals count", ax=ax2, legend=False, color="r", figsize=(20, 4))
    ax.figure.legend()
